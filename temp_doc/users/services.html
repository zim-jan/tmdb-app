 23    def register_user(
 24        self,
 25        username: str,
 26        email: str,
 27        nickname: str,
 28        password: str,
 29    ) -> User:
 30        """
 31        Register a new user.
 32
 33        Parameters
 34        ----------
 35        username : str
 36            Desired username.
 37        email : str
 38            User's email address.
 39        nickname : str
 40            User's public nickname.
 41        password : str
 42            User's password (will be hashed).
 43
 44        Returns
 45        -------
 46        User
 47            Created user object.
 48
 49        Raises
 50        ------
 51        ValueError
 52            If username, email, or nickname already exists.
 53        """
 54        # Check if username exists
 55        if User.objects.filter(username=username).exists():
 56            raise ValueError("Username already exists")
 57
 58        # Check if email exists
 59        if User.objects.filter(email=email).exists():
 60            raise ValueError("Email already exists")
 61
 62        # Check if nickname exists
 63        if User.objects.filter(nickname=nickname).exists():
 64            raise ValueError("Nickname already exists")
 65
 66        # Create user
 67        user = User.objects.create_user(
 68            username=username,
 69            email=email,
 70            password=password,
 71        )
 72        user.nickname = nickname
 73        user.save()
 74
 75        return user

 77    def authenticate_user(self, username: str, password: str) -> User | None:
 78        """
 79        Authenticate a user.
 80
 81        Parameters
 82        ----------
 83        username : str
 84            Username or email.
 85        password : str
 86            User's password.
 87
 88        Returns
 89        -------
 90        User | None
 91            Authenticated user or None if authentication failed.
 92        """
 93        # Try to authenticate with username
 94        user = authenticate(username=username, password=password)
 95
 96        # If failed, try with email
 97        if user is None:
 98            try:
 99                user_obj = User.objects.get(email=username)
100                user = authenticate(username=user_obj.username, password=password)
101            except User.DoesNotExist:
102                pass
103
104        return user # pyright: ignore[reportReturnType]

111        nickname: str | None = None,

113        """
114        Update user profile information.

116        Parameters

123            New nickname (optional).

127        User
128            Updated user object.

130        Raises
131        ------

141            if User.objects.filter(nickname=nickname).exists():
142                raise ValueError("Nickname already exists")
143            user.nickname = nickname

145        user.save()
146        return user
