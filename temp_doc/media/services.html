 33    def create_media_from_tmdb(self, tmdb_id: int, media_type: str) -> Media:
 34        """
 35        Create media object from TMDb data.
 36
 37        Parameters
 38        ----------
 39        tmdb_id : int
 40            TMDb identifier for the media.
 41        media_type : str
 42            Type of media ("MOVIE" or "TV_SHOW").
 43
 44        Returns
 45        -------
 46        Media
 47            Created media object.
 48
 49        Raises
 50        ------
 51        ValueError
 52            If media_type is invalid or TMDb data is unavailable.
 53        """
 54        # Check if media already exists
 55        existing_media = Media.objects.filter(tmdb_id=tmdb_id, media_type=media_type).first()
 56        if existing_media:
 57            return existing_media
 58
 59        # Fetch data from TMDb
 60        if media_type == "MOVIE":
 61            tmdb_data = self.tmdb_service.get_movie_details(tmdb_id)
 62        elif media_type == "TV_SHOW":
 63            tmdb_data = self.tmdb_service.get_tv_details(tmdb_id)
 64        else:
 65            raise ValueError(f"Invalid media type: {media_type}")
 66
 67        # Get appropriate factory
 68        factory = MediaFactoryProvider.get_factory(media_type)
 69
 70        # Prepare common data
 71        media_data = self._parse_tmdb_data(tmdb_data, media_type)
 72
 73        # Create media using factory
 74        return factory.create_media(**media_data)

 76    def _parse_tmdb_data(self, tmdb_data: dict[str, Any], media_type: str) -> dict[str, Any]:
 77        """
 78        Parse TMDb API data into model fields.
 79
 80        Parameters
 81        ----------
 82        tmdb_data : dict[str, Any]
 83            Raw data from TMDb API.
 84        media_type : str
 85            Type of media ("MOVIE" or "TV_SHOW").
 86
 87        Returns
 88        -------
 89        dict[str, Any]
 90            Parsed data ready for model creation.
 91        """
 92        # Parse release date
 93        release_date_str = tmdb_data.get("release_date") or tmdb_data.get("first_air_date")
 94        release_date = None
 95        if release_date_str:
 96            with contextlib.suppress(ValueError):
 97                release_date = datetime.strptime(release_date_str, "%Y-%m-%d").date()
 98
 99        # Common fields
100        data: dict[str, Any] = {
101            "tmdb_id": tmdb_data["id"],
102            "title": tmdb_data.get("title") or tmdb_data.get("name", ""),
103            "original_title": tmdb_data.get("original_title") or tmdb_data.get("original_name", ""),
104            "overview": tmdb_data.get("overview", ""),
105            "poster_path": tmdb_data.get("poster_path", ""),
106            "backdrop_path": tmdb_data.get("backdrop_path", ""),
107            "release_date": release_date,
108            "popularity": tmdb_data.get("popularity", 0.0),
109            "vote_average": tmdb_data.get("vote_average", 0.0),
110            "vote_count": tmdb_data.get("vote_count", 0),
111            "original_language": tmdb_data.get("original_language", ""),
112        }
113
114        # Movie-specific fields
115        if media_type == "MOVIE":
116            data["runtime"] = tmdb_data.get("runtime")
117            data["budget"] = tmdb_data.get("budget", 0)
118            data["revenue"] = tmdb_data.get("revenue", 0)
119
120        # TV Show-specific fields
121        elif media_type == "TV_SHOW":
122            data["number_of_seasons"] = tmdb_data.get("number_of_seasons", 0)
123            data["number_of_episodes"] = tmdb_data.get("number_of_episodes", 0)
124
125            episode_run_times = tmdb_data.get("episode_run_time", [])
126            data["episode_run_time"] = episode_run_times[0] if episode_run_times else None
127
128            data["status"] = tmdb_data.get("status", "")
129
130            # Parse dates
131            first_air_date_str = tmdb_data.get("first_air_date")
132            if first_air_date_str:
133                try:
134                    data["first_air_date"] = datetime.strptime(first_air_date_str, "%Y-%m-%d").date()
135                except ValueError:
136                    data["first_air_date"] = None
137
138            last_air_date_str = tmdb_data.get("last_air_date")
139            if last_air_date_str:
140                try:
141                    data["last_air_date"] = datetime.strptime(last_air_date_str, "%Y-%m-%d").date()
142                except ValueError:
143                    data["last_air_date"] = None
144
145        return data

147    def search_media(self, query: str, media_type: str | None = None, enrich: bool = True) -> list[dict[str, Any]]:
148        """
149        Search for media by title.
150
151        Parameters
152        ----------
153        query : str
154            Search query.
155        media_type : str | None
156            Type of media to search ("movie", "tv", or None for both).
157        enrich : bool
158            Whether to enrich results with additional details (credits, images).
159
160        Returns
161        -------
162        list[dict[str, Any]]
163            List of search results from TMDb.
164        """
165        results: list[dict[str, Any]] = []
166        
167        # Normalize media_type to uppercase format
168        normalized_type = None
169        if media_type:
170            if media_type.lower() == "movie":
171                normalized_type = "MOVIE"
172            elif media_type.lower() == "tv":
173                normalized_type = "TV_SHOW"
174
175        if normalized_type is None or normalized_type == "MOVIE":
176            movie_results = self.tmdb_service.search_movie(query)
177            for result in movie_results:
178                result["media_type"] = "movie"
179                result["title"] = result.get("title", "")
180                result["rating"] = result.get("vote_average", 0)
181                # Enrich with credits and images if requested
182                if enrich:
183                    result = self.tmdb_service.enrich_search_result(result, "movie")
184            results.extend(movie_results)
185
186        if normalized_type is None or normalized_type == "TV_SHOW":
187            tv_results = self.tmdb_service.search_tv_show(query)
188            for result in tv_results:
189                result["media_type"] = "tv"
190                result["title"] = result.get("name", "")
191                result["rating"] = result.get("vote_average", 0)
192                # Enrich with credits and images if requested
193                if enrich:
194                    result = self.tmdb_service.enrich_search_result(result, "tv")
195            results.extend(tv_results)
196
197        return results

201        """
202        Update media metadata from TMDb.
203
204        Parameters
205        ----------
206        media : Media
207            Media object to update.
208
209        Returns
210        -------
211        Media
212            Updated media object.
213        """
214        # Fetch fresh data from TMDb
215        tmdb_data = self.tmdb_service.get_movie_details(media.tmdb_id) if media.media_type == "MOVIE" else self.tmdb_service.get_tv_details(media.tmdb_id)
216
217        # Parse and update

219
220        for field, value in parsed_data.items():
221            setattr(media, field, value)
222
223        media.save()
224        return media
